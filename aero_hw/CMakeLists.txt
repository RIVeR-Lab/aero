cmake_minimum_required(VERSION 2.8.3)
project(aero_hw)

find_package(catkin REQUIRED COMPONENTS
  control_toolbox
  controller_manager
  hardware_interface
  roscpp
  river_ros_util
  transmission_interface
  device_driver_base
  roboteq_driver
)

catkin_package(
  INCLUDE_DIRS include
  LIBRARIES 
  CATKIN_DEPENDS control_toolbox controller_manager hardware_interface roscpp river_ros_util transmission_interface device_driver_base roboteq_driver
  DEPENDS 
)

include_directories(
  ${catkin_INCLUDE_DIRS}
  include
)


add_executable(aero_base_hw_node
	src/aero_base_hw_node.cpp)
target_link_libraries(aero_base_hw_node
  ${catkin_LIBRARIES}
)
